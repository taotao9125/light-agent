     │                                                          │
     │  1. POST /login { username, password }                   │
     │ ─────────────────────────────────────────────────────▶  │
     │                                                          │
     │                                                          │ 2. 验证账号密码
     │                                                          │    ↓

     │                                                          │    jwt.sign({ userId: 123 }, '密钥')

     │  4. 返回 { token: "eyJhbGciOiJ..." }                    │
     │ ◀─────────────────────────────────────────────────────  │
     │                                                          │
     │ 5. 存储 JWT                                              │
     │    - localStorage / sessionStorage                       │

     │                                                          │
     │                          ┌──────────────┐                │
     │                          │   登录完成    │                │
     │                          └──────────────┘                │


┌─────────────────────────────────────────────────────────────────────────────┐
│                            第二次：访问受保护资源                            │
└─────────────────────────────────────────────────────────────────────────────┘

客户端（浏览器）                                              服务端
     │                                                          │
     │  1. 从存储中取出 JWT                                      │
     │    ↓                                                     │
     │  2. 放在请求头中                                          │
     │    Authorization: Bearer eyJhbGciOi...                   │
     │                                                          │
     │  3. GET /profile + 请求头                                │
     │ ─────────────────────────────────────────────────────▶  │
     │                                                          │
     │                                                          │ 4. 提取 JWT
     │                                                          │    从 Authorization 头取出
     │                                                          │    ↓
     │                                                          │ 5. 验证签名
     │                                                          │    jwt.verify(token, '密钥')
     │                                                          │    ↓
     │                                          ┌──────────────┐│
     │                                          │ 验证失败？    ││
     │                                          └──────────────┘│
     │                                            │            │
     │                          ┌─────────────────┴────────────┐│
     │                          │ 失败                          │ 成功
     │                          ▼                              ▼│
     │                          │                    6. 从 token 取出用户信息
     │                          │                       decoded.userId = 123
     │                          │                    7. 处理业务逻辑
     │                          │                              │
     │  8. 返回 401             │                    8. 返回数据 │
     │ ◀─────────────────────────────────────────────────────  │
     │                          │                              │
     │  9. 跳转登录页           │    9. 显示用户信息            │
     │                          │                              │
